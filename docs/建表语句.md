-- ==========================================
-- 0. 拆迁队入场：清空旧表 (CASCADE 会自动解除外键关联)
-- ==========================================
DROP TABLE IF EXISTS learning_progress CASCADE;
DROP TABLE IF EXISTS qa_records CASCADE;
DROP TABLE IF EXISTS cir_sections CASCADE;
DROP TABLE IF EXISTS lessons CASCADE;
DROP TABLE IF EXISTS courses CASCADE;
DROP TABLE IF EXISTS course_categories CASCADE;
DROP TABLE IF EXISTS users CASCADE;

-- ==========================================
-- 1. 基础信息域：对接泛雅平台与身份管理
-- ==========================================

-- 用户表：支持画像(UserProfile)能力，提升AI对齐理解程度
CREATE TABLE users (
    user_id VARCHAR(50) PRIMARY KEY,
    school_id VARCHAR(50) NOT NULL, -- 多租户与机构数据隔离
    grade VARCHAR(20) NOT NULL, -- 年级，如“大一/2024级”
    major VARCHAR(100) NOT NULL, -- 专业
    phone_number VARCHAR(20) UNIQUE,
    user_name VARCHAR(100) NOT NULL,
    role VARCHAR(20) CHECK (role IN ('teacher', 'student')),
    contact_info JSONB,
    profile_text TEXT, -- 用户补充说明（学习基础、目标等）
    profile_ext JSONB, -- 结构化画像扩展（兴趣、偏好交互模式等）
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 课程分类表：大类折叠菜单
CREATE TABLE course_categories (
    category_id SERIAL PRIMARY KEY,
    school_id VARCHAR(50) NOT NULL, -- 隔离
    category_name VARCHAR(100) NOT NULL,
    sort_order INT DEFAULT 0
);

-- 课程体系表
CREATE TABLE courses (
    course_id VARCHAR(50) PRIMARY KEY, 
    school_id VARCHAR(50) NOT NULL, -- 隔离
    category_id INT REFERENCES course_categories(category_id),
    course_name VARCHAR(200) NOT NULL, 
    teacher_id VARCHAR(50) REFERENCES users(user_id),
    term VARCHAR(20),
    ext_info JSONB
);

-- ==========================================
-- 2. 智课生成域：异步解析与 CIR 目录引擎 (核心)
-- ==========================================

-- 课件任务表：监控解析状态与性能
CREATE TABLE lessons (
    lesson_id VARCHAR(50) PRIMARY KEY, -- 对应 parseId
    course_id VARCHAR(50) REFERENCES courses(course_id),
    school_id VARCHAR(50) NOT NULL, -- 隔离
    title VARCHAR(200) NOT NULL,
    cover_image VARCHAR(500),
    file_type VARCHAR(10),
    file_url VARCHAR(500) NOT NULL,
    category VARCHAR(20),
    task_status VARCHAR(20) DEFAULT 'processing',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    file_info JSONB
);

-- CIR 课程中间结构表：统一 node_id，支持前端高亮与路径检索
CREATE TABLE cir_sections (
    node_id VARCHAR(50) PRIMARY KEY, -- 统一命名升级 (原 section_id)
    lesson_id VARCHAR(50) REFERENCES lessons(lesson_id),
    school_id VARCHAR(50) NOT NULL,
    node_name VARCHAR(200) NOT NULL, -- 节点标题 (原 section_name)
    
    parent_id VARCHAR(50) REFERENCES cir_sections(node_id), -- 指向父节点
    node_type VARCHAR(20) CHECK (node_type IN ('chapter', 'subchapter', 'point')), 
    sort_order INT DEFAULT 0,
    
    path TEXT, -- 节点路径（如 /chapter1/section2/point3），用于检索对齐
    page_num INT, -- 对应PPT页码 (原 page_index)
    image_url VARCHAR(500), -- 页面图片 (原 ppt_image_url)
    bbox JSONB, -- 选填：答案高亮定位框 ([x, y, w, h])
    
    key_points JSONB,
    teaching_content TEXT,
    script_content TEXT,
    duration_seconds INT,
    audio_url VARCHAR(500),
    media_resources JSONB,
    is_teacher_edited BOOLEAN DEFAULT FALSE
);

-- ==========================================
-- 3. 互动与续接域：问答交互与进度适配
-- ==========================================

-- 问答记录表：支持多来源证据数组与思考过程追踪
CREATE TABLE qa_records (
    answer_id SERIAL PRIMARY KEY,
    session_id VARCHAR(50) NOT NULL, -- 相同会话代表多轮追问
    user_id VARCHAR(50) REFERENCES users(user_id),
    school_id VARCHAR(50) NOT NULL,
    lesson_id VARCHAR(50) REFERENCES lessons(lesson_id), -- 允许为空（聊天阶段）
    
    question_type VARCHAR(10),
    question_text TEXT NOT NULL,
    answer_text TEXT NOT NULL,
    
    -- 证据溯源核心升级
    cited_node_id VARCHAR(50) REFERENCES cir_sections(node_id), -- 主引用节点
    source_page_num INT, -- 主引用页码
    sources JSONB, -- 多来源证据数组，包含 bbox, score 等前端高亮信息
    
    current_path TEXT, -- 提问时的学习路径
    video_timestamp DOUBLE PRECISION, -- 播放器时间戳(秒)，用于精准续接
    
    understanding_level VARCHAR(20),
    response_ms INT,
    is_accurate BOOLEAN,
    
    -- AI 调试与过程数据
    reasoning_content TEXT, -- reasoner 的思考过程
    tool_calls JSONB, -- 工具调用记录 (web_search等)
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 学习进度与适配表：承接后端状态机与掌握度逻辑
CREATE TABLE learning_progress (
    track_id VARCHAR(50) PRIMARY KEY,
    user_id VARCHAR(50) REFERENCES users(user_id),
    session_id VARCHAR(50) NOT NULL, -- 会话对齐
    school_id VARCHAR(50) NOT NULL,
    lesson_id VARCHAR(50) REFERENCES lessons(lesson_id),
    
    current_node_id VARCHAR(50) REFERENCES cir_sections(node_id), -- 断点节点
    current_path TEXT, -- 当前学习路径
    current_topic TEXT, -- 当前主题
    last_qa_query TEXT, -- 最近一次有效问答
    
    confusion_count INT DEFAULT 0, -- 困惑计数，触发退回/补讲
    mastery JSONB DEFAULT '{}', -- 主题掌握度映射 (如 {"极限": 0.73})
    
    last_position_seconds INT DEFAULT 0,
    progress_percent FLOAT,
    
    adjust_type VARCHAR(20) DEFAULT 'normal',
    needs_supplement BOOLEAN DEFAULT FALSE,
    last_operate_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
